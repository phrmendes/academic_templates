% indentation macros
\newcommand{\nid}{\noindent}
\newcommand{\zeroIndent}{\setlength{\parindent}{0cm}}
\newcommand{\oneHalfIndent}{\setlength{\parindent}{1.25cm}}

% reference style macro
\newcommand{\referenceStyle}{
  \pagestyle{plain}
  \small
  \singlespacing
  \zeroIndent
}

% figure macro
\newcommand{\insertFigure}[5]{
  \vspace{0.5cm}
  \begin{figure}[H]
    \centering
    \caption{#1}
    \includegraphics[#2\textwidth]{#3}
    \label{fig:#4}

    \centering \footnotesize Fonte: #5
  \end{figure}
  \vspace{0.5cm}
}

% image reference macro
\newcommand{\refBegin}{
  \centering
  \footnotesize
}

\newcommand{\refEnd}{
  \normalsize
  \justify
}

% quote macro
\newcommand{\qtBegin}{
  \small
  \singlespacing
  \zeroIndent
}

\newcommand{\qtEnd}{
  \onehalfspacing
  \normalsize
  \vspace{0.5cm}
  \oneHalfIndent
}
